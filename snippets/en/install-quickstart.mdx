<Tabs>
  <Tab title="Homebrew">
    ```bash
    brew tap diillson/chatcli && brew install chatcli
    ```
  </Tab>
  <Tab title="go install">
    ```bash
    go install github.com/diillson/chatcli@latest
    ```
  </Tab>
  <Tab title="Docker">
    ```bash
    docker pull ghcr.io/diillson/chatcli:latest
    ```
  </Tab>
  <Tab title="Helm">
    ```bash
    helm install chatcli oci://ghcr.io/diillson/charts/chatcli
    helm install chatcli-operator oci://ghcr.io/diillson/charts/chatcli-operator
    ```
  </Tab>
  <Tab title="Binary">
    Download the binary from [Releases](https://github.com/diillson/chatcli/releases) and:

    ```bash
    chmod +x chatcli && sudo mv chatcli /usr/local/bin/
    ```
  </Tab>
</Tabs>
