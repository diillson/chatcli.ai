---
title: "Create Runbook"
api: "POST /api/v1/runbooks"
description: "Creates a new remediation runbook"
icon: "plus"
---

<ParamField body="name" type="string" required>
  Unique runbook name (slug, e.g., `runbook-disk-cleanup`)
</ParamField>

<ParamField body="category" type="string" required>
  Category: `oomkill`, `crashloop`, `latency`, `scaling`, `disk`, `network`, `custom`
</ParamField>

<ParamField body="description" type="string" required>
  Description of the runbook's objective
</ParamField>

<ParamField body="severity" type="string[]" required>
  List of target severities: `critical`, `high`, `medium`, `low`
</ParamField>

<ParamField body="automated" type="boolean" default="false">
  Whether the runbook can be executed automatically
</ParamField>

<ParamField body="approvalRequired" type="boolean" default="true">
  Whether approval is required before execution
</ParamField>

<ParamField body="rollbackOnFailure" type="boolean" default="true">
  Whether to perform automatic rollback on failure
</ParamField>

<ParamField body="steps" type="object[]" required>
  List of runbook steps
  <Expandable title="Step properties">
    <ParamField body="name" type="string" required>
      Unique step identifier
    </ParamField>
    <ParamField body="description" type="string" required>
      Step description
    </ParamField>
    <ParamField body="action" type="string" required>
      Action type: `kubectl`, `script`, `ai-analyze`, `notification`, `wait`
    </ParamField>
    <ParamField body="command" type="string">
      Command to execute (for `kubectl` and `script` actions)
    </ParamField>
    <ParamField body="timeout" type="string" default="60s">
      Step timeout in Go duration format
    </ParamField>
    <ParamField body="continueOnFailure" type="boolean" default="false">
      Whether to continue to the next step even on failure
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="matchConditions" type="object">
  Conditions for automatic matching with incidents
  <Expandable title="Properties">
    <ParamField body="resourceTypes" type="string[]">
      Resource types (e.g., `["Deployment", "StatefulSet"]`)
    </ParamField>
    <ParamField body="descriptionPatterns" type="string[]">
      Regex patterns for matching against incident descriptions
    </ParamField>
    <ParamField body="namespaces" type="string[]">
      Namespaces where the runbook applies
    </ParamField>
  </Expandable>
</ParamField>
