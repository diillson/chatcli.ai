---
title: "List PostMortems"
api: "GET /api/v1/postmortems"
description: "Returns the list of postmortems automatically generated by the platform"
icon: "list"
---

<ParamField query="incident" type="string">
  Filter by associated incident (e.g., `INC-20260319-001`)
</ParamField>

<ParamField query="severity" type="string">
  Filter by original incident severity: `critical`, `high`, `medium`, `low`
</ParamField>

<ParamField query="since" type="string">
  Filter postmortems created from a given ISO 8601 date
</ParamField>

<ParamField query="page" type="integer" default="1">
  Page number
</ParamField>

<ParamField query="pageSize" type="integer" default="20">
  Items per page (maximum 100)
</ParamField>
