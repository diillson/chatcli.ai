---
title: "Update Runbook"
api: "PUT /api/v1/runbooks/{name}"
description: "Updates an existing remediation runbook"
---

<ParamField path="name" type="string" required>
  Unique runbook name (e.g., `runbook-oomkill-standard`)
</ParamField>

<ParamField body="description" type="string">
  New runbook description
</ParamField>

<ParamField body="severity" type="string[]">
  Updated list of target severities
</ParamField>

<ParamField body="automated" type="boolean">
  Whether the runbook can be executed automatically
</ParamField>

<ParamField body="approvalRequired" type="boolean">
  Whether approval is required before execution
</ParamField>

<ParamField body="rollbackOnFailure" type="boolean">
  Whether to perform automatic rollback on failure
</ParamField>

<ParamField body="steps" type="object[]">
  Updated list of runbook steps
  <Expandable title="Step properties">
    <ParamField body="name" type="string" required>
      Unique step identifier
    </ParamField>
    <ParamField body="description" type="string" required>
      Step description
    </ParamField>
    <ParamField body="action" type="string" required>
      Action type: `kubectl`, `script`, `ai-analyze`, `notification`, `wait`
    </ParamField>
    <ParamField body="command" type="string">
      Command to execute
    </ParamField>
    <ParamField body="timeout" type="string" default="60s">
      Step timeout
    </ParamField>
    <ParamField body="continueOnFailure" type="boolean" default="false">
      Whether to continue on failure
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="matchConditions" type="object">
  Updated conditions for automatic matching
  <Expandable title="Properties">
    <ParamField body="resourceTypes" type="string[]">
      Resource types
    </ParamField>
    <ParamField body="descriptionPatterns" type="string[]">
      Regex patterns for matching
    </ParamField>
    <ParamField body="namespaces" type="string[]">
      Applicable namespaces
    </ParamField>
  </Expandable>
</ParamField>
