---
title: "Analytics Trends"
api: "GET /api/v1/analytics/trends"
description: "Returns issue trends over time with severity breakdown — useful for identifying patterns and measuring operational health"
icon: "arrow-trend-up"
---

<ParamField query="from" type="string">
  Start date in RFC3339 format
</ParamField>

<ParamField query="to" type="string">
  End date in RFC3339 format
</ParamField>

<ParamField query="groupBy" type="string" default="day">
  Group results by `day` or `week`
</ParamField>
