---
title: "Atualizar Runbook"
api: "PUT /api/v1/runbooks/{name}"
description: "Atualiza um runbook de remediação existente"
---

<ParamField path="name" type="string" required>
  Nome único do runbook (ex: `runbook-oomkill-standard`)
</ParamField>

<ParamField body="description" type="string">
  Nova descrição do runbook
</ParamField>

<ParamField body="severity" type="string[]">
  Lista atualizada de severidades alvo
</ParamField>

<ParamField body="automated" type="boolean">
  Se o runbook pode ser executado automaticamente
</ParamField>

<ParamField body="approvalRequired" type="boolean">
  Se requer aprovação antes da execução
</ParamField>

<ParamField body="rollbackOnFailure" type="boolean">
  Se deve executar rollback automático em caso de falha
</ParamField>

<ParamField body="steps" type="object[]">
  Lista atualizada de passos do runbook
  <Expandable title="Propriedades do step">
    <ParamField body="name" type="string" required>
      Identificador único do passo
    </ParamField>
    <ParamField body="description" type="string" required>
      Descrição do passo
    </ParamField>
    <ParamField body="action" type="string" required>
      Tipo de acao: `kubectl`, `script`, `ai-analyze`, `notification`, `wait`
    </ParamField>
    <ParamField body="command" type="string">
      Comando a ser executado
    </ParamField>
    <ParamField body="timeout" type="string" default="60s">
      Timeout do passo
    </ParamField>
    <ParamField body="continueOnFailure" type="boolean" default="false">
      Se deve continuar em caso de falha
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="matchConditions" type="object">
  Condicoes atualizadas para matching automatico
  <Expandable title="Propriedades">
    <ParamField body="resourceTypes" type="string[]">
      Tipos de recurso
    </ParamField>
    <ParamField body="descriptionPatterns" type="string[]">
      Padroes regex para match
    </ParamField>
    <ParamField body="namespaces" type="string[]">
      Namespaces aplicaveis
    </ParamField>
  </Expandable>
</ParamField>
