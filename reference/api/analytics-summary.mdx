---
title: "Resumo de Analytics"
api: "GET /api/v1/analytics/summary"
description: "Retorna um resumo consolidado de métricas operacionais da plataforma"

</ParamField>

<ParamField query="namespace" type="string">

</ParamField>

<ParamField query="cluster" type="string">
  Filtrar por cluster (em ambientes federados)
</ParamField>
