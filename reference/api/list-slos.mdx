---
title: "Listar SLOs"
api: "GET /api/v1/slos"
description: "Retorna a lista de SLOs (Service Level Objectives) configurados"
---

<ParamField query="namespace" type="string">
  Filtrar por namespace Kubernetes
</ParamField>

<ParamField query="service" type="string">
  Filtrar por nome do serviço
</ParamField>

<ParamField query="status" type="string">
  Filtrar por status: `healthy`, `warning`, `breached`
</ParamField>

<ParamField query="page" type="integer" default="1">
  Número da página
</ParamField>

<ParamField query="pageSize" type="integer" default="20">
  Itens por página (máximo 100)
</ParamField>
